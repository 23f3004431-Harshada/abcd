NPV Calculation — Python


def npv(cashflows, rate):
    """
    cashflows: list where cashflows[0] = t0 (initial), rate in decimal (e.g., 0.08)
    """
    return sum(cf / ((1 + rate) ** t) for t, cf in enumerate(cashflows))

# Example usage
if __name__ == "__main__":
    cashflows = [-100000, 30000, 40000, 40000, 35000]
    print("NPV:", npv(cashflows, 0.08))
